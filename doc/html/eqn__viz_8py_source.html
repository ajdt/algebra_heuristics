528  def splitPredicate(pred_string):
529  """return predicate name and list of operands from predicate string"""
530  # remove all whitespace first
531  pred_string = ''.join(pred_string.split())
532  open_paren = pred_string.find('(')
533  close_paren = pred_string.rfind(')')
534 
535  if open_paren == -1 or close_paren == -1:
536  return (pred_string, [])
537 
538  pred_name = pred_string[:open_paren]
539  operand_str = pred_string[open_paren+1:close_paren]
540  # split operands into an array
541  start_idx = 0
542  paren_level = 0
543 
544  operands = []
545  for index in range(0, len(operand_str)):
546  if operand_str[index] == '(':
547  paren_level += 1
548  elif operand_str[index] == ')':
549  paren_level -= 1
550  elif operand_str[index] == ',' and paren_level == 0:
551  operands.append(operand_str[start_idx:index])
552  start_idx = index + 1
553  continue
554 
555  # add the last operand
556  if index == len(operand_str) - 1 and paren_level == 0:
557  operands.append(operand_str[start_idx:])
558 
559  return (pred_name, operands)
